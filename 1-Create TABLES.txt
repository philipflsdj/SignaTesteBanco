Create Database Signa 

use Signa;

CREATE TABLE Owner(
Id INT IDENTITY(1,1) NOT NULL PRIMARY KEY,
Name VARCHAR(300)
)

CREATE TABLE Cats(
Id INT IDENTITY(1,1) NOT NULL PRIMARY KEY,
Name VARCHAR(300),
Age INT,
Ownerid INT NOT NULL,
FOREIGN KEY(Ownerid) REFERENCES Owner(Id) 
)

CREATE TABLE Dogs(
Id INT IDENTITY(1,1) NOT NULL PRIMARY KEY,
Name VARCHAR(300),
Age INT,
Ownerid INT NOT NULL,
FOREIGN KEY(Ownerid) REFERENCES Owner(Id) 
)
